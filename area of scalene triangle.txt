//C Program to Calculate Area of Scalene Triangle



#include <stdio.h>

int main() {
   float a=0.5,b,h,area;
   
   printf("Enter the value of base and height ");
   scanf("%f %f",&b,&h);
   
   area=a*b*h;
   printf("Area of scalene triangle = %f",area);

    return 0;
}
